    "    \"\"\"昇温に必要なステップ数を計算\"\"\"\n",
    "    delta_T = T_end - T_start\n",
    "    heating_time_ps = delta_T / heating_rate  # [ps]\n",
    "    heating_time_fs = heating_time_ps * 1000  # [fs]\n",
    "    n_steps = int(heating_time_fs / dt_fs)\n",
    "    return n_steps, heating_time_ps\n",
    "\n",
    "def calculate_equil_steps(equil_time_ps, dt_fs=2.0):\n",
    "    \"\"\"平衡化に必要なステップ数を計算\"\"\"\n",
    "    equil_time_fs = equil_time_ps * 1000  # [fs]\n",
    "    n_steps = int(equil_time_fs / dt_fs)\n",
    "    return n_steps\n",
    "\n",
    "# 各フェーズのステップ数を計算\n",
    "n_steps_heat_room, heat_time_room = calculate_heating_steps(T_INIT, T_ROOM, HEATING_RATE)\n",

    "n_steps_heat_target, heat_time_target = calculate_heating_steps(T_ROOM, T_TARGET, HEATING_RATE)\n",

    "pfactor = PFACTOR_BASE**2 * DT**2\n",
    "\n",
    "# 計算時間の見積もり\n",
    "total_heating_time = heat_time_room + heat_time_target\n",
    "total_equil_time = EQUIL_TIME_PS * 2\n",

    "    print(f\"  >> Plot saved to: {img_filename}\")\n",
    "    \n",
    "    # ドリフトチェック\n",
    "    half_idx = len(volumes) // 2\n",
    "    if half_idx > 0:\n",
    "        vol_drift = np.mean(volumes[half_idx:]) - np.mean(volumes[:half_idx])\n",
    "        print(f\"  >> Volume Drift: {vol_drift:.4f} A^3\")\n",
    "        \n",

    "class PropertyAverager:\n",
    "    def __init__(self, atoms):\n",
    "        self.atoms = atoms\n",
    "        self.reset()\n",
    "    def reset(self):\n",
    "        self.cells = []\n",
    "        self.volumes = []\n",
    "        self.pot_energies = []\n",
    "    def sample(self):\n",
    "        self.cells.append(self.atoms.cell.cellpar())\n",
    "        self.volumes.append(self.atoms.get_volume())\n",
    "        self.pot_energies.append(self.atoms.get_potential_energy())\n",
    "    def print_average(self, label, temperature):\n",
    "        if not self.volumes:\n",
    "            print(f\"No data for {label}\")\n",
    "            return\n",
    "        avg_cell = np.mean(self.cells, axis=0)\n",
    "        avg_vol = np.mean(self.volumes)\n",
    "        avg_pe = np.mean(self.pot_energies)\n",
    "        total_mass = sum(self.atoms.get_masses())\n",
    "        avg_dens = (total_mass / avg_vol) * 1.66054\n",
    "        print(f\"\\n=== {label} (Avg @ {temperature}K) ===\")\n",
    "        print(f\"  Lattice: {avg_cell[:3]}\")\n",
    "        print(f\"  Angles : {avg_cell[3:]}\")\n",
    "        print(f\"  Volume : {avg_vol:.4f} A^3\")\n",
    "        print(f\"  Density: {avg_dens:.4f} g/cm^3\")\n",
    "        print(\"========================================\\n\")\n",
    "\n",
